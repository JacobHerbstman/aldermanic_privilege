        \vspace{0.5em}
        \textbf{Permits are the binding constraint}
        \begin{itemize}
            \item Permitting is the primary driver of supply inelasticity \citep{soltas_how_2026}
            \item Focus on \textbf{high-discretion} permits: new construction, renovation, demolition
            \item High-discretion processing times are \textbf{negatively} correlated with permit volume; low-discretion times are \textbf{positively} correlated \hyperlink{permit-vol-corr-appendix}{\beamergotobutton{Permit Volume}}

% SLIDE: STRINGENCY INDEX CONSTRUCTION
% ==============================================================================
\begin{frame}
	\frametitle{Stringency Index Construction}
	\begin{enumerate}
		\item \textbf{Residualize} processing times on permit-level characteristics and a rich set of fixed effects (permit type, review type, and year $\times$ month)
		\begin{itemize}
			\item Controls for baseline differences across wards (income, demographics, etc.)
		\end{itemize}
		\item \textbf{Estimate alderman fixed effects} on residualized processing times
		\[
			\widehat{\epsilon}_{it} = \mu + \delta_a + \eta_{it}
		\]
		\vspace{-0.5em}
		\item \textbf{Empirical Bayes shrinkage} to reduce noise {\scriptsize \citep{kline_systemic_2022,kline_discrimination_2024}} then standardize to mean 0, SD 1
	\end{enumerate}
	
	\vspace{0.5em}
	\textbf{Interpretation}: High score = alderman associated with longer processing times
	
	\vspace{0.3em}
	\textbf{Caveat}: This is a \textit{proxy}: likely understates true variation in bureaucratic frictions 
	\begin{itemize}
		\item Misses projects that never apply or are discouraged early
	\end{itemize}
\end{frame}
