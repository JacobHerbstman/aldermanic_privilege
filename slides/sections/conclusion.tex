\begin{frame}
    \frametitle{Discussion \& Policy Implications}
    
    \textbf{Core problem}: Aldermen don't internalize citywide benefits of density
    \begin{itemize}
        \item Agglomeration spillovers extend beyond ward boundaries
        \item Negative externalities (congestion) are local
        \item $\Rightarrow$ Systematic under-provision of housing relative to the social optimum 
    \end{itemize}
    
    \vspace{0.5em}
    \textbf{Policy alternatives to centralize decision-making}:
    \begin{enumerate}
        \item \textbf{Transfer decision-making authority}: Shift decision-making to the citywide level (i.e. mayor, zoning board)
        \item \textbf{Citywide preemption}: Override local vetoes for certain housing types \\
              {\small (e.g., CA SB-35, MA Ch. 40B, federal H.O.M.E. Act proposals)}
        \item \textbf{Less aldermen}: Reduce number of wards to increase average ward size and internalize more spillovers
    \end{enumerate}
\end{frame}

\begin{frame}
    \frametitle{Conclusion}
    
    \textbf{Summary of findings}:
    \begin{enumerate}
        \item Aldermen vary substantially in how they exercise zoning discretion
        \item Stricter aldermen $\Rightarrow$ 10-16\% lower multifamily density at borders
        \item Stricter aldermen $\Rightarrow$ 3-5\% higher rents, 2\% higher home prices
    \end{enumerate}
    
    \vspace{0.5em}
    \textbf{Implications}:
    \begin{itemize}
        \item Decentralized zoning creates real welfare costs
        \item Within-city fragmentation matters, not just across municipalities
        \item Reforms should consider who bears costs vs.\ who makes decisions
    \end{itemize}
    
    \vspace{0.5em}
    \textbf{Future work}:
    \begin{itemize}
        \item Structural estimation of agglomeration spillover decay
        \item Welfare analysis of alternative governance structures
    \end{itemize}
\end{frame}
