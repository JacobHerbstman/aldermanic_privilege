\begin{frame}
    \frametitle{Discussion \& Policy Implications}
    
    \textbf{Core problem}: Regulatory stringency and discretion raise the cost of housing development and lead to lower density
    \begin{itemize}
        \item Longer permitting times + more uncertainty $\rightarrow$ higher carrying costs for developers
        \item Fewer projects initiated $\rightarrow$ constrained rental supply $\rightarrow$ higher rents
        \item Effects stronger in the rental market, but appears to spill over to housing market as well
    \end{itemize}

    \vspace{0.5em}
    \textbf{Policy alternatives to reduce regulatory friction}:
    \begin{enumerate}
        \item \textbf{Reduce Individual Discretion}: Remove aldermanic privilege and ability to change zoning or delay permits when development is by-right
        \item \textbf{Transfer decision-making authority}: Shift zoning decisions to the citywide level (i.e.\ mayor, zoning board) to reduce fragmentation and local capture
    \end{enumerate}
\end{frame}

% ==============================================================================
% SLIDE: CONCLUSION
% ==============================================================================
\begin{frame}
    \frametitle{Conclusion}
    
    \textbf{Summary of findings}:
    \begin{enumerate}
        \item More stringent aldermen $\Rightarrow$ lower multifamily density at ward borders by about $10\%$
        \item About $3-4\%$ higher rents, which appears to be driven by reduced supply
    \end{enumerate}

    \vspace{0.5em}
    \textbf{Implications}:
    \begin{itemize}
        \item Regulatory stringency in decentralized zoning lowers supply and raises rents
        \item Within-city fragmentation matters, not just across municipalities
        \item Reforms should target decision-making authority and discretion, not just by-right zoning reforms
    \end{itemize}

\end{frame}
