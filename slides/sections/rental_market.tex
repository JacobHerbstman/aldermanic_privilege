\begin{frame}
    \frametitle{From Density to Rents}

    \textbf{So far}: More stringent aldermen $\rightarrow$ lower multifamily density

    \vspace{0.5em}
    \textbf{Next question}: Does this supply constraint feed into higher housing costs?

    \vspace{0.5em}
    \textbf{Approach}: Apply the same border-pair FE design to rental listings
    \begin{itemize}
        \item Data: RentHub rental listings, 2014-2022
        \item 2.71 million listing observations, 141 ward pairs, BW = 500 ft
        \item Segment $\times$ year-month FE; SEs clustered at ward-pair level
    \end{itemize}
\end{frame}

% SLIDE: RULING OUT QUALITY SORTING
% ==============================================================================
\begin{frame}
    \frametitle{Ruling Out Quality Sorting}

    \textbf{Concern}: Rent premium could reflect \textit{better} units on the more stringent side

    \vspace{0.3em}
    \textbf{Test}: Same border-pair FE on unit characteristics (BW = 500 ft, 2014--2022)

    \vspace{0.3em}
    \begin{center}
    \scriptsize
    \input{../tasks/rental_characteristics_at_borders/output/char_fe_table_bw500_clust_ward_pair.tex}
    \end{center}

    \vspace{0.3em}
    \small
    $\Rightarrow$ Core size/bed/bath margins are close to zero; composition shifts are small. \\
    Quality sorting is unlikely to explain the full rent premium at ward boundaries.
\end{frame}


\begin{frame}
    \frametitle{Ruling Out Amenity Sorting}

    \textbf{Concern}: Rent premium could reflect \textit{better neighborhood amenities} on the more stringent side

    \vspace{0.2em}
    \textbf{Test}: Same border-pair FE on amenity access (BW = 500 ft, 2014--2022)

    \vspace{0.2em}
    \begin{center}
    \resizebox{\linewidth}{!}{\input{../tasks/rental_characteristics_at_borders/output/amenity_char_fe_table_bw500_pre_2023_clust_ward_pair.tex}}
    \end{center}

    \vspace{0.2em}
    \small
    $\Rightarrow$ Distances to schools, parks, major roads, and the lake are imprecisely estimated and close to zero. \\
    Large amenity differences are unlikely to drive the rent premium.
\end{frame}
