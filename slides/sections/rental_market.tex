\begin{frame}
    \frametitle{From Density to Rents}

    \textbf{So far}: More stringent aldermen $\rightarrow$ lower multifamily density

    \vspace{0.5em}
    \textbf{Next question}: Does this supply constraint feed into higher housing costs?

    \vspace{0.5em}
    \textbf{Approach}: Apply the same ward-boundary RD design to rental listings and home sales
    \begin{itemize}
        \item RentHub rental listings, 2014--2022
        \item Cook County home sales, 2006--2022
        \item Segment-time fixed effects compare nearby observations at the same ward boundary
    \end{itemize}
\end{frame}
