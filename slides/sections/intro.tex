\begin{frame}
	\frametitle{Motivation}
	\begin{itemize}
		\item Housing prices and rents are skyrocketing in many U.S.\ cities, including Chicago
		\begin{itemize}
			\item Limited housing supply in high demand areas is widely accepted as the key driver of these price increases
			\item Land use regulations and local control are often blamed for limiting supply
		\end{itemize}
		\item Most research focuses on formal institutions such as by-right zoning laws
		\begin{itemize}
			\item But regulatory discretion and stringency are also key determinants of housing supply and are less studied
		\end{itemize}
		\item \textbf{Chicago}: Unique informal institution of \textit{aldermanic privilege}
		\begin{itemize}
			\item Offers a unique setting where de facto regulatory environment changes discretely at ward boundaries
		\end{itemize}
	\end{itemize}
\end{frame}

% ==============================================================================
% SLIDE 4: WHAT IS ALDERMANIC PRIVILEGE?
% ==============================================================================
\begin{frame}
	\frametitle{What is Aldermanic Privilege?}
	\begin{itemize}
		\item \textbf{City council logistics}
		\begin{itemize}
			\item 50 wards + aldermen, each voted on only by the residents of their ward. 
			\item Each ward has a population of about 50,000 people.
			\item 4 year terms, but aldermen are often in office for decades 
		\end{itemize}
		\item \textbf{Core feature: Deference}
		\begin{itemize}
			\item Aldermanic privilege is not written in any law but is universally acknowledged
			\item ``Very few things happen that aldermen don't want to happen.'' \mbox{\scriptsize \citep{zhang_boundaries_2011}}
			\item Often referred to as ``little mayors'' ruling wards as ``fiefdoms''
		\end{itemize}
		\item \textbf{Chicago gives more power to local politicians than many other U.S.\ cities}: 
		\begin{itemize}
			\item Houston has no explicit zoning code and centralized development approvals
			\item Minneapolis has reduced councilmember discretion in recent years
			\item Both cities have seen below-median rent and home price growth compared to peer cities \cite{hembre_unpacking_2025,heckler_this_2026}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{comment}
\begin{frame}
	\frametitle{How Zoning Works in Chicago}
	\textbf{Formal Process:}
	\begin{enumerate}
		\item Developer wants more density than as-of-right $\rightarrow$ requests rezone
		\item Application goes to Zoning Administrator $\rightarrow$ Dept.\ of Planning $\rightarrow$ Committee on Zoning $\rightarrow$ City Council vote
	\end{enumerate}
	
	\vspace{0.1em}
	\textbf{Informal Reality:}
	\begin{itemize}
		\item Alderman consulted at \textit{every} step (even though not formally required)
		\item Many aldermen have their own ward application forms
		\item Alderman says yes $\rightarrow$ passes unanimously
		\item Alderman says no $\rightarrow$ dead on arrival (colleagues defer, often no city council vote)
	\end{itemize}
	\textbf{Result:}
	\begin{itemize}
		\item De facto veto power creates a tremendous amount of stringency for developers
	\end{itemize}
\end{frame}
\end{comment}

% SLIDE 6: THIS PAPER
% ==============================================================================
\begin{frame}
	\frametitle{This Paper}
	\textbf{Research Question}: Does heterogeneity in regulatory stringency reduce housing supply and raise rents?

	\vspace{0.5em}
	\textbf{Empirical strategy}: Border discontinuity design at ward boundaries
	\begin{itemize}
		\item Look at multifamily development and rents near ward borders 
		\item \textbf{Key assumption}: neighborhood characteristics vary smoothly across the border but the regulatory environment changes discretely
	\end{itemize}

	\vspace{0.3em}
	\textbf{Preview of Results}
	\begin{enumerate}
		\item More stringent aldermen $\Rightarrow$ lowers multifamily density near ward borders
		\item $\sim$3\% higher rents on the more stringent side of the border, smaller effects on home prices
		\item Effect appears to operate through reduced supply
	\end{enumerate}
\end{frame}
