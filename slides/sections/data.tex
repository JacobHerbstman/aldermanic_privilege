    \frametitle{Data Sources}
	\begin{itemize}
        % Item 1: Appears on slide 1 and stays visible.
        \item<1-> \textbf{Cook County Assessor Open Data}
        % Details for item 1: Only appear on slide 1.
        \only<1>{
            \begin{itemize} 
               \item Primary data source is the universe of parcel assessments, 1999-2023
               \item Calculate distances to ward borders using geolocated parcels dataset, matched on pin10
               \item Control variables such as amenities come from the "Parcel Proximity" dataset
            \end{itemize}
        }
        \vspace{1em} 

        \item<2-> \textbf{Other Data Sources I can use}
        % Details for item 1: Only appear on slide 1.
        \only<2>{
            \begin{itemize} 
                \item Used in other tasks: new residential construction data from "Single and Multi-Family Improvement Characteristics"
                \item Gives me units, stories, density metrics such as Floor Area Ratio (FAR)/ Lot Area Per Unit (LAPU)
                \item Building Permits Data as another source for calculating alderman strictness
            \end{itemize}
        }
        \vspace{1em} 

        % Item 3: Appears on slide 2 and stays visible.
        \item<3-> \textbf{Shapefiles}
        % Details for item 2: Only appear on slide 3.
        \only<3>{
            \begin{itemize}
                \item I get 2003-2014 and 2015-2023 ward boundaries from the city open data portal
                \item 1998-2003 shapefiles available from UIUC, haven't incorporated into event study yet 
            \end{itemize}
        }
    \end{itemize}
\end{frame}
