        \textbf{Why both measures?}
        
        \vspace{0.1em}
        \resizebox{\linewidth}{!}{$\mathrm{FAR}=\frac{\text{floor area}}{\text{lot area}} \qquad \mathrm{DUPAC}=\frac{\text{\# dwelling units}}{\text{lot acreage}}$}

        \vspace{0.1em}
        \begin{itemize}\setlength{\itemsep}{1pt}\setlength{\topsep}{2pt}
            \item \textbf{Developers} care about FAR (total rentable space)
            \item \textbf{Neighbors} care about DUPAC (congestion, parking, schools)
        \end{itemize}

\begin{frame}[label=identification-strategy]
    \frametitle{Identification Strategy: Border Discontinuity}

    \textbf{Idea}: Compare new multifamily construction on opposite sides of ward boundary segments
    \begin{itemize}
        \item Focus on parcels \textbf{within 500 feet} of a segment
        \item Fixed effects: zoning group, boundary segment, and year
    \end{itemize}

    \vspace{0.3em}
    \textbf{Identifying variation}: Within the same boundary segment, controlling for zoning group and citywide year fixed effects, does the more stringent side have less dense new multifamily construction? \hyperlink{border-segments}{\beamergotobutton{Border Segments}}

    \vspace{0.3em}
    Two specifications:
    \begin{itemize}
        \item \textbf{Visual RD}: FE-adjusted discontinuity plots
        \item \textbf{Regressions}: FE regressions using continuous variation in the stringency index
    \end{itemize}
\end{frame}
