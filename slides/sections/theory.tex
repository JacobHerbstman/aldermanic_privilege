\begin{frame}
	\frametitle{A ``Time-to-Build'' Model of Regulatory Stringency}
	\small
	\textbf{Key idea}: Regulatory environment $\rightarrow$ effective delay $T(s)$ $\rightarrow$ discounts developer revenue

	\vspace{0.5em}
	\textbf{Developer's problem.} A project chooses density $q \geq 0$ with cost $C(q) = F + cq + \frac{\gamma}{2}q^2$.
	\vspace{0.2em}

	Revenue arrives upon completion, discounted by regulatory delay:
	\vspace{-0.3em}
	\[
		\max_{q \geq 0} \quad e^{-rT(s)}\, p\, q \;-\; C(q)
	\]
	\vspace{-0.8em}

	where $r$ = discount rate, $T(s)$ = time-to-build under regulatory environment $s$

	\vspace{0.5em}
	\textbf{Optimal density} (interior solution):
	\vspace{-0.3em}
	\[
		q^*(T, p) = \frac{e^{-rT(s)}\, p - c}{\gamma} \qquad \text{if } e^{-rT(s)} p > c, \;\text{else } q^* = 0
	\]
	\vspace{-0.5em}

	\textbf{Sufficient statistic}: $rT(s)$ summarizes the regulatory wedge --- captures delay, redesign, negotiation, financing frictions, and regulatory uncertainty
\end{frame}

\begin{frame}
	\frametitle{Testable Predictions}
	\small
	\textbf{Core comparative static}: $\;\partial q^* / \partial T(s) < 0$, which implies:
	\vspace{0.3em}
	\begin{enumerate}
		\item \textbf{Lower density} --- completed projects have fewer units \hfill {\scriptsize\textcolor{gray}{$\rightarrow$ Border-pair FE}}
		\item \textbf{Higher prices/rents} --- lower aggregate supply $Q = \sum_i q_i^{*}$, $\;p'(Q) < 0$ \hfill {\scriptsize\textcolor{gray}{$\rightarrow$ Event study}}
		%\item \textbf{Amplified by financing costs} --- effects scale with $r$ \hfill {\scriptsize\textcolor{gray}{$\rightarrow$ Heterogeneity}}
		%\item \textbf{Selection} --- surviving projects are higher-return on observables
	\end{enumerate}

	\vspace{0.5em}
	\textbf{Mapping to data}:
	\begin{itemize}
		\item $T(s)$ proxied by observed permit-to-completion times
		\item Aldermanic strictness score $\approx$ alderman-specific component of $T(s)$
		\item Cross-boundary variation: different aldermen $\Rightarrow$ different $T(s)$ on each side
	\end{itemize}
\end{frame}
