    \end{column}
    
    \begin{column}{0.43\textwidth}
        \small
        \textbf{DUPAC captures population density}
        
        \vspace{0.5em}
        $\text{DUPAC} = \dfrac{\text{\# Dwelling Units}}{\text{Lot Acreage}}$
        
        \vspace{0.5em}
        \textbf{Why both measures?}
        \begin{itemize}
            \item Same FAR, different DUPAC
            \item \textbf{Developers} care about FAR (total rentable space)
            \item \textbf{Neighbors} care about DUPAC (congestion, parking, schools)
        \end{itemize}
    \end{column}
    \end{columns}

% SLIDE 15: IDENTIFICATION STRATEGY
% ==============================================================================
\begin{frame}
    \frametitle{Identification Strategy: Border Discontinuity}
    
    \textbf{Idea}: Compare new multifamily construction on opposite sides of ward boundary segments
    \begin{itemize}
        \item Focus on parcels \textbf{within 500 feet} of a segment
        \item Fixed effects: zoning code, boundary segment, and year
    \end{itemize}
    
    \vspace{0.3em}
    \textbf{Identifying variation}: Within the same boundary segment, controlling for zoning code and citywide year fixed effects, does the more stringent side have less dense new multifamily construction?
    
    \vspace{0.3em}
    Two specifications:
    \begin{itemize}
        \item \textbf{Visual RD}: FE-adjusted discontinuity plots
        \item \textbf{Regressions}: FE regressions using continuous variation in the stringency index
    \end{itemize}
\end{frame}
