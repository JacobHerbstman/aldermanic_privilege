% SLIDE: FROM DENSITY TO RENTS
% ==============================================================================
\begin{frame}
    \frametitle{From Density to Rents}

    \textbf{So far}: More uncertain aldermen $\rightarrow$ lower multifamily density

    \vspace{0.5em}
    \textbf{Next question}: Does this supply constraint feed into higher housing costs?

    \vspace{0.5em}
    \textbf{Approach}: Apply the same border-pair FE design to rental listings
    \begin{itemize}
        \item Data: RentHub rental listings, 2014-2022
        \item 2.67 million listing observations, 141 ward pairs, BW = 500 ft
        \item Ward-pair $\times$ year-month FE; SEs clustered at ward-pair level
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Ruling Out Quality Sorting}

    \textbf{Concern}: Rent premium could reflect \textit{better} units on the more uncertain side

    \vspace{0.3em}
    \textbf{Test}: Same border-pair FE on unit characteristics (BW = 500 ft, 2014--2022)

    \vspace{0.3em}
    \begin{center}
    \scriptsize
    \input{../tasks/rental_characteristics_at_borders/output/char_fe_table_bw500.tex}
    \end{center}

    \vspace{0.3em}
    \small
    $\Rightarrow$ Core size/bed/bath margins are close to zero; composition shifts are small. \\
    Quality sorting is unlikely to explain the full rent premium at ward boundaries.
\end{frame}

        \small
        \textbf{Outcome}: Log(distinct listings) per ward-pair $\times$ side $\times$ year-month cell

        \vspace{0.5em}
        $\Rightarrow$ $\sim$10\% fewer distinct rental listings on the more uncertain side

        \vspace{0.5em}
        \textbf{Chain}: Regulatory uncertainty $\rightarrow$ fewer units built $\rightarrow$ fewer available rentals $\rightarrow$ higher rents
    \end{column}
    \end{columns}
